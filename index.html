let steps = 0

	if (n == 1) {
		console.log
			(`Move Disk1 from ${fromStick} to ${toStick}`)
		steps++
		return 0
	}

	tower(n - 1, fromStick, usingStick, toStick, steps++)

	console.log
		(`Move Disk ${n} from ${fromStick} to ${toStick}`)

	tower(n - 1, usingStick, toStick, fromStick, steps++)

}
